getWeeklyMaxWeight(logs: ExerciseLogEntry[], exerciseId: string, weekMonday: string): number | null
  - Filter logs by exerciseId AND date >= weekMonday AND date < next monday
  - If filtered is empty: return null
  - TWO-BRANCH LOGIC (D-07):
    Compute: const hasWeight = filtered.some(l => l.setsData?.some(s => s.weightKg !== undefined))
    (a) Weight branch (hasWeight === true): return max of all set.weightKg values across filtered entries
        (ignore undefined/0 values; return 0 if no positive values found)
    (b) Bodyweight branch (hasWeight === false, all entries are bodyweight-only): return max of all

getTrendBadge(currentWeekMax: number | null, prevWeekMax: number | null): '↑' | '–' | '↓' | null
  - currentWeekMax === null → null (hidden)
  - currentWeekMax > prevWeekMax (or prevWeekMax === null and currentWeekMax > 0) → '↑'
  - currentWeekMax === prevWeekMax → '–'
  - currentWeekMax < prevWeekMax → '↓'

getTrendColor(badge: '↑' | '–' | '↓' | null): string
  - '↑' → Colors.status.optimal (green)
  - '↓' → Colors.status.critical (red)
  - '–' → Colors.onSurfaceMuted (grey)
  - null → 'transparent'

getLastSessionSummary(logs: ExerciseLogEntry[], exerciseId: string): string | null
