```typescript
// All functions are async and return typed results from @supabase/supabase-js

// Sign up: creates email account. For anonymous sessions, uses updateUser
// to promote the anonymous session in-place (converts it to email/password).
// Returns { user, error }
export async function signUpWithEmail(
  email: string,
  password: string
): Promise<{ user: import('@supabase/supabase-js').User | null; error: string | null }>

// Sign in: authenticates existing email/password account.
// Returns { user, error }
export async function signInWithEmail(
  email: string,
  password: string
): Promise<{ user: import('@supabase/supabase-js').User | null; error: string | null }>

// Convert anonymous session to email account in-place.
// Uses supabase.auth.updateUser({ email, password }) — this is the Supabase-approved
// method for anonymous→email conversion (NOT linkIdentity which is OAuth-only).
// Call this after signUpWithEmail confirms the user was anonymous (user.is_anonymous).
// Returns { user, error }
export async function convertAnonymousToEmail(
  email: string,
  password: string
): Promise<{ user: import('@supabase/supabase-js').User | null; error: string | null }>

// Sends a Supabase password reset email to the given address.
// Returns { error: string | null }
export async function sendPasswordResetEmail(
  email: string
): Promise<{ error: string | null }>

// Signs out the current user (clears Supabase session token).
// Does NOT touch AsyncStorage keys — per D-08.
// Returns { error: string | null }
export async function signOutUser(): Promise<{ error: string | null }>

// Resends the email verification/confirmation email to the current user's address.
// Uses supabase.auth.resend({ type: 'signup', email })
// Returns { error: string | null }
export async function resendVerificationEmail(
  email: string
): Promise<{ error: string | null }>

// Maps a raw Supabase AuthError message to a user-facing string per D-15.
// Called by all screens that display auth errors.
export function mapAuthError(message: string): string
// Error mapping (D-15):
//   message contains 'Invalid login credentials' or 'wrong password' → "Incorrect password"
//   message contains 'User not found' or 'No user found' → "No account found with that email"
//   message contains 'Network request failed' or 'fetch' (network) → "No internet connection — please try again"
//   message contains 'rate limit' or 'too many' → "Too many attempts — please wait a few minutes"
//   message contains 'Email not confirmed' or 'not confirmed' → "Please verify your email first"
//   fallback → "Something went wrong — try again"
```
