```typescript
const init = async () => {
  try {
    await initSupabaseSession();
    const { data: { user } } = await supabase.auth.getUser();
    if (user && !user.is_anonymous) {
      setInitialRoute('Main');
    } else {
      const raw = await AsyncStorage.getItem('@vitalspan_user_profile').catch(() => null);
      // onboardingComplete check is NOT needed here — WelcomeScreen handles it
      // Any anonymous/no-session user always sees Welcome; CTAs handle Onboarding routing
      setInitialRoute('Welcome');
    }
  } catch {
    setInitialRoute('Welcome');
  }
  // Fire-and-forget: cache pruning + data migration (non-blocking)
  pruneExpiredCache().catch(() => null);
  // Migration chain: only runs after session is established
  void (async () => {
    const migrated = await AsyncStorage.getItem('@vitalspan_migrated_v2').catch(() => null);
    if (!migrated) {
      const biomarkersRaw = await AsyncStorage.getItem('@vitalspan_biomarkers').catch(() => null);
      const entries = biomarkersRaw ? JSON.parse(biomarkersRaw) : [];
      migrateHistory(entries)
        .then(() => AsyncStorage.setItem('@vitalspan_migrated_v2', 'true'))
        .catch(() => null);
    }
  })();
};
```

Note on the replacement init():
- `@vitalspan_user_profile` read is removed from routing — per D-06, anonymous/no-session always → Welcome.

- `pruneExpiredCache()` remains fire-and-forget but moves AFTER the try/catch block
  (routing is set first, then non-blocking cleanup runs).
- Migration chain remains non-blocking and still fires regardless of session type.
- The `raw` variable inside the else branch is intentionally unused — the read is a no-op guard;
  remove it entirely if the linter complains: all anonymous users go to Welcome unconditionally.
